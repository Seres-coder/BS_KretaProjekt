  TIMESTAMPTZ '2025-09-01 08:00:00+02' + ((d.diak_id * 5 + t.tantargy_id * 2 + g.sorszam) * INTERVAL '1 day') AS datum,
  TIMESTAMPTZ '2025-09-01 08:00:00+02' + ((d.diak_id * 5 + t.tantargy_id * 2 + g.sorszam + 1) * INTERVAL '1 day') AS updatedatum,
  ((d.diak_id + t.tantargy_id + g.sorszam) % 5) + 1 AS ertek,
  t.tantargy_id,
  t.tantargy_id AS tanar_id,
  d.diak_id
FROM "Diakok" d
CROSS JOIN "Tantargyok" t
CROSS JOIN (VALUES (1), (2), (3)) AS g(sorszam)
ORDER BY d.diak_id, t.tantargy_id, g.sorszam;
